### 合约部署流程
```solidity
pragma solidity ^0.5.4;

contract Token {
    mapping (address => uint) public balances;
    
    function transfer(address to, uint amount) external {
        require(balances[msg.sender] >= amount);
        balances[msg.sender] -= amount;
        balances[to] += amount;
    }
}
```
